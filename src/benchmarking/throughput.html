Running Load Tests

bash
# Install k6
brew install k6  # macOS
# or: https://k6.io/docs/getting-started/installation/

# Run throughput test
k6 run --vus 500 --duration 60s benchmarks/k6/throughput.js

# Run with mocked provider
BIFROST_MOCK=true k6 run --vus 1000 --duration 120s benchmarks/k6/throughput.js
