Riven AI/Get Started/Java SDK

Java SDK

Enterprise-grade Java SDK with reactive streams and virtual threads support.

The Java SDK provides a comprehensive Java interface to Riven AI with support for both traditional blocking calls and modern reactive streams. Compatible with Java 17+ and supports virtual threads (Project Loom) for optimal scalability.

Features

Reactive Streams

Built-in support for Project Reactor and RxJava for reactive applications.

Virtual Threads

Java 21 virtual threads support for high-concurrency applications.

Spring Boot Integration

Auto-configuration for seamless Spring Boot integration.

Jackson & Gson Support

Flexible JSON serialization with your preferred library.

Installation

Add the Riven AI SDK to your Maven or Gradle project:

pom.xml
<dependency>
    <groupId>ai.rivenai</groupId>
    <artifactId>riven-sdk</artifactId>
    <version>0.1.0</version>
</dependency>

Or with Gradle:

build.gradle
implementation 'ai.rivenai:riven-sdk:0.1.0'

Quick Start

Here is a minimal example to get started with the Java SDK:

Main.java
1import ai.riven.sdk.Riven AI;
2import ai.riven.sdk.models.*;
3
4public class Main {
5    public static void main(String[] args) {
6        // Initialize the client
7        RivenClient client = RivenClient.builder()
8            .baseUrl("http://localhost:8080")
9            .apiKey("your-virtual-key") // Optional
10            .build();
11
12        // Make a chat completion request
13        ChatCompletionResponse response = riven.chat().completions().create(
14            ChatCompletionRequest.builder()
15                .provider("openai")
16                .model("gpt-4o-mini")
17                .messages(List.of(
18                    ChatMessage.user("Hello, how are you?")
19                ))
20                .maxTokens(100)
21                .temperature(0.7)
22                .build()
23        );
24
25        System.out.println(response.getChoices().get(0).getMessage().getContent());
26    }
27}

Async with CompletableFuture

Use CompletableFuture for non-blocking operations:

AsyncExample.java
1import ai.riven.sdk.Riven AI;
2import ai.riven.sdk.models.*;
3import java.util.concurrent.CompletableFuture;
4
5public class AsyncExample {
6    public static void main(String[] args) {
7        RivenClient client = RivenClient.builder()
8            .baseUrl("http://localhost:8080")
9            .build();
10
11        CompletableFuture<ChatCompletionResponse> future = riven.chat()
12            .completions()
13            .createAsync(
14                ChatCompletionRequest.builder()
15                    .provider("openai")
16                    .model("gpt-4o-mini")
17                    .messages(List.of(
18                        ChatMessage.user("Write a haiku about Java")
19                    ))
20                    .build()
21            );
22
23        future.thenAccept(response -> {
24            System.out.println(response.getChoices().get(0).getMessage().getContent());
25        }).exceptionally(e -> {
26            System.err.println("Error: " + e.getMessage());
27            return null;
28        }).join();
29    }
30}

Streaming Responses

Use reactive streams for real-time streaming responses:

StreamingExample.java
1import ai.riven.sdk.Riven AI;
2import ai.riven.sdk.models.*;
3import reactor.core.publisher.Flux;
4
5public class StreamingExample {
6    public static void main(String[] args) {
7        RivenClient client = RivenClient.builder()
8            .baseUrl("http://localhost:8080")
9            .build();
10
11        Flux<ChatCompletionChunk> stream = riven.chat()
12            .completions()
13            .createStream(
14                ChatCompletionRequest.builder()
15                    .provider("openai")
16                    .model("gpt-4o-mini")
17                    .messages(List.of(
18                        ChatMessage.user("Write a short poem about coding")
19                    ))
20                    .maxTokens(500)
21                    .stream(true)
22                    .build()
23            );
24
25        stream.subscribe(
26            chunk -> {
27                String content = chunk.getChoices().get(0).getDelta().getContent();
28                if (content != null) {
29                    System.out.print(content);
30                }
31            },
32            error -> System.err.println("Error: " + error.getMessage()),
33            () -> System.out.println()
34        );
35
36        // Block until complete
37        stream.blockLast();
38    }
39}

Using Fallbacks

Configure automatic failover between providers:

FallbacksExample.java
1import ai.riven.sdk.Riven AI;
2import ai.riven.sdk.models.*;
3
4public class FallbacksExample {
5    public static void main(String[] args) {
6        RivenClient client = RivenClient.builder()
7            .baseUrl("http://localhost:8080")
8            .build();
9
10        ChatCompletionResponse response = riven.chat()
11            .completions()
12            .create(
13                ChatCompletionRequest.builder()
14                    .provider("openai")
15                    .model("gpt-4o-mini")
16                    .messages(List.of(
17                        ChatMessage.user("Explain quantum computing")
18                    ))
19                    .maxTokens(1000)
20                    .fallbacks(List.of(
21                        Fallback.of("anthropic", "claude-3-5-sonnet-20241022"),
22                        Fallback.of("bedrock", "anthropic.claude-3-sonnet-20240229-v1:0")
23                    ))
24                    .build()
25            );
26
27        // Check which provider actually served the request
28        System.out.printf("Response from %s: %s%n",
29            response.getProvider(),
30            response.getChoices().get(0).getMessage().getContent()
31        );
32    }
33}

Spring Boot Integration

Use the Spring Boot starter for auto-configuration:

pom.xml
<dependency>
    <groupId>ai.rivenai</groupId>
    <artifactId>riven-spring-boot-starter</artifactId>
    <version>0.1.0</version>
</dependency>
application.yml
riven:
  base-url: http://localhost:8080
  api-key: ${RIVEN_API_KEY:}
ChatController.java
1import ai.riven.sdk.Riven AI;
2import ai.riven.sdk.models.*;
3import org.springframework.web.bind.annotation.*;
4import reactor.core.publisher.Flux;
5
6@RestController
7@RequestMapping("/api")
8public class ChatController {
9
10    private final Riven AI riven;
11
12    public ChatController(Riven AI riven) {
13        this.riven = riven;
14    }
15
16    @PostMapping("/chat")
17    public ChatResponse chat(@RequestBody ChatRequest request) {
18        ChatCompletionResponse response = riven.chat()
19            .completions()
20            .create(
21                ChatCompletionRequest.builder()
22                    .provider("openai")
23                    .model("gpt-4o-mini")
24                    .messages(List.of(
25                        ChatMessage.user(request.getMessage())
26                    ))
27                    .build()
28            );
29
30        return new ChatResponse(
31            response.getChoices().get(0).getMessage().getContent()
32        );
33    }
34
35    @PostMapping(value = "/chat/stream", produces = "text/event-stream")
36    public Flux<String> chatStream(@RequestBody ChatRequest request) {
37        return riven.chat()
38            .completions()
39            .createStream(
40                ChatCompletionRequest.builder()
41                    .provider("openai")
42                    .model("gpt-4o-mini")
43                    .messages(List.of(
44                        ChatMessage.user(request.getMessage())
45                    ))
46                    .stream(true)
47                    .build()
48            )
49            .mapNotNull(chunk -> chunk.getChoices().get(0).getDelta().getContent());
50    }
51}
